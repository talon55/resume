% Using this space to override existing commands
\newbox{\cventrymainbox} % create box to be measured
\newlength{\cventrymainlength} % create main length variable
\newlength{\cventrycitylength} % create city length variable
\renewcommand*{\cventry}[7][.25em]{%
  % Define box based on 3rd and optional 4th arguments
  \savebox{\cventrymainbox}{{\bfseries#3}
    \ifthenelse{\equal{#4}{}}{}{, {\slshape#4}.}} % end box
  % set main length based on size of box
  \setlength{\cventrymainlength}{\widthof{\usebox{\cventrymainbox}}}
  % Set city length based on space leftover
  \setlength{\cventrycitylength}{\maincolumnwidth-\separatorcolumnwidth-\cventrymainlength}

  \cvitem[#1]{#2}{%
    \begin{minipage}[t]{\cventrymainlength}
    {\bfseries#3}% bfseries causes all following text to be boldface
    \ifthenelse{\equal{#4}{}}{}{, {\slshape#4}}% slshape causes all following text to be italicized
    \end{minipage}
    \ifthenelse{\equal{#5}{}}{}{\begin{minipage}[t]{\cventrycitylength}\raggedleft#5\end{minipage}}%
    \ifthenelse{\equal{#6}{}}{}{\newline #6}%
    \strut%
    \ifx&#7&%
      \else{\newline{}\begin{minipage}[t]{\linewidth}\small#7\end{minipage}}\fi}}
